const express = require("express"); const bodyParser = require("body-parser"); const fs = require("fs"); const app = express(); app.use(bodyParser.json()); let db = { users: [], posts: [] }; // Charger données if (fs.existsSync("db.json")) { db = JSON.parse(fs.readFileSync("db.json")); } // Sauvegarde function saveDB() { fs.writeFileSync("db.json", JSON.stringify(db, null, 2)); } // -------- AUTH -------- app.post("/register", (req, res) => { const { username, password } = req.body; if (db.users.find(u => u.username === username)) { return res.send("Utilisateur existe déjà"); } db.users.push({ username, password }); saveDB(); res.send("Compte créé"); }); app.post("/login", (req, res) => { const { username, password } = req.body; const user = db.users.find(u => u.username === username && u.password === password); if (!user) return res.send("Erreur login"); res.send("OK"); }); // -------- POSTS -------- app.get("/posts", (req, res) => { res.json(db.posts); }); app.post("/posts", (req, res) => { const { user, content } = req.body; const post = { id: Date.now(), user, content, likes: 0, comments: [] }; db.posts.unshift(post); saveDB(); res.json(post); }); // -------- LIKE -------- app.post("/like/:id", (req, res) => { const post = db.posts.find(p => p.id == req.params.id); if (post) { post.likes++; saveDB(); } res.send("liked"); }); // -------- COMMENT -------- app.post("/comment/:id", (req, res) => { const post = db.posts.find(p => p.id == req.params.id); if (post) { post.comments.push(req.body.comment); saveDB(); } res.send("commented"); }); // -------- FRONTEND -------- app.get("/", (req, res) => { res.send(` Mini Facebook Pro

Mini Social App

`); }); // -------- START -------- app.listen(3000, () => { console.log("Serveur lancé sur http://localhost:3000"); });
